def fit_model(X, y):
    """  [X,y]"""
    
    cross_validator = KFold()
    
    regressor = DecisionTreeRegressor()

    params = { 'max_depth': np.arange(10) + 1 }

    #  [X,y]
    grid = grid.fit(X, y)

    #
